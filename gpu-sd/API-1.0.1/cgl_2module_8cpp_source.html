00057     std::deque< GPUInfo > infos;

00059     {

00063         info.device = i;
00064         info.pvp[0] = int32_t(displayRect.origin.x);
00065         info.pvp[1] = int32_t(displayRect.origin.y);
00066         info.pvp[2] = int32_t(displayRect.size.width);
00067         info.pvp[3] = int32_t(displayRect.size.height);
00068 
00069         if( mainDisplayID == displayIDs[i] )
00070             infos.push_front( info );
00071         else
00072             infos.push_back( info );
00073     }
00074
