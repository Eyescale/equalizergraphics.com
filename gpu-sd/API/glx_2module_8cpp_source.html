00040 {
00041     int major, event, error;

00049         const std::string displayNumber = displayString.substr( colonPos+1 );

00060     if( vendor && std::string( vendor ) == "VirtualGL" )

00062         info.flags |= GPUInfo::FLAG_VIRTUALGL;
00063         const char* vglDisplay = getenv( "VGL_DISPLAY" );

00065         std::stringstream xDisplay;
00066

00069             xDisplay << "." << DefaultScreen( display );
00070 
00071         if( vglDisplayStr == xDisplay.str( ))
00072             info.flags |= GPUInfo::FLAG_VIRTUALGL_DISPLAY;

00124     // try x servers :0 - :n
00125     for( unsigned i = 0; i < std::numeric_limits< unsigned >::max() ; ++i )
00126         // x screens :n.0 - :n.m

00129             std::stringstream stream;

00131                 
00132             GPUInfo info( "GLX" );
00133             if( queryDisplay_( stream.str(), info ))
00134             {
00135                 if( info != defaultInfo )
00136                     result.push_back( info );

00139                 // X Server does not exist, stop query
